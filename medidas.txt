Valor Comparativo Dinâmico = 
VAR _indicadorSelecionado = SELECTEDVALUE('Categorias Comparativo'[Indicador])
VAR _periodoSelecionado = SELECTEDVALUE('Legenda Anual'[Período])
RETURN
    SWITCH(
        TRUE(),
        _indicadorSelecionado = "Venda Bruta" && _periodoSelecionado = "Ano Atual", [Venda Bruta Ano Selecionado],
        _indicadorSelecionado = "Venda Bruta" && _periodoSelecionado = "Ano Anterior", [Venda Bruta Ano Anterior],
        _indicadorSelecionado = "Vendas Líquidas" && _periodoSelecionado = "Ano Atual", [Venda Líquida Ano Selecionado],
        _indicadorSelecionado = "Vendas Líquidas" && _periodoSelecionado = "Ano Anterior", [Venda Líquida Ano Anterior],
        _indicadorSelecionado = "Gross" && _periodoSelecionado = "Ano Atual", [Gross Ano Selecionado],
        _indicadorSelecionado = "Gross" && _periodoSelecionado = "Ano Anterior", [Gross Ano Anterior],
        _indicadorSelecionado = "Churn" && _periodoSelecionado = "Ano Atual", [Churn Ano Selecionado],
        _indicadorSelecionado = "Churn" && _periodoSelecionado = "Ano Anterior", [Churn Ano Anterior],
        _indicadorSelecionado = "Netadds" && _periodoSelecionado = "Ano Atual", [Netadds Ano Selecionado],
        _indicadorSelecionado = "Netadds" && _periodoSelecionado = "Ano Anterior", [Netadds Ano Anterior],
        BLANK()
    )
